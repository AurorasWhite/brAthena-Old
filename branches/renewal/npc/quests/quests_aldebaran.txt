aldeba_in,152,166,4	script	Comerciante#01	86,{ 
        mes "[Comerciante]"; 
        mes "Quem é este homem misterioso?"; 
        mes "Eu, o enigmático e charmoso 'Comerciante?'"; 
        next; 
        mes "[Comerciante]"; 
        mes "Viajando pelo continente de Rune-Midgard, com todos os seus chapéus maravilhosos? Ninguém sabe..."; 
        next; 
        mes "[Comerciante]"; 
        mes "Para que propósito posso colecionar e trocar esses chapéus de todo o mundo? Escolha um chapéu, vão desvendando o mistério..."; 
        next; 
        mes " (1) ^3355FFEspelho de Médico^000000:"; 
        mes "1 Faixa Marrom + 50 Ferros + 1 Diamante Rachado + 3500 Zeny"; 
        mes " (2)^3355FFChapéu com Pluma^000000:"; 
        mes "1 Chapéu Romântico + 300 Plumas + 500 Zeny"; 
        mes " (3) ^3355FFMáscara do Fantasma^000000:"; 
        mes "20 Ferro + 1 Planta da Melodia + 5000 Zeny "; 
        mes " (4) ^3355FFChapéu Chinês^000000:"; 
        mes "120 Trunks + 10000 Zeny "; 
        next; 
        switch(select(" Espelho de Médico : Chapéu com Pluma : Phantom of Opera : Sakkat ")) { 
        case 1: 
                if ((countitem(2275) > 0) && (countitem(998) > 49) && (countitem(733) > 0) && (Zeny > 3499)) { 
                        delitem 2275,1;  // Faixa Marrom 
                        delitem 998,50;  // Ferro 
                        delitem 733,1;  // Diamante Rachado 
                        set Zeny,Zeny-3500; 
                        mes "[Comerciante]"; 
                        mes "Hm! Você não tem uma licença médica, não é? Está tudo bem, eu ouvi sobre um Gatuno, o médico não licenciado que realizou milagres médicos! Mas... Isso pode ter sido uma história em quadrinhos."; 
                        next; 
                        mes "[Comerciante]"; 
                        mes "Oh que seja. Só não seja pego."; 
                        getitem 2273,1; // Doctor_Cap, Espelho de Médico 
                        close; 
                } 
                else { 
                        mes "[Comerciante]"; 
                        mes "Você cara~"; 
                        mes "Verifique os requisitos de novo."; 
                        mes "Você não pareçe um idiota, mas. Então cara, caia na real."; 
                        close; 
                } 
        case 2: 
                if ((countitem(2247) > 0) && (countitem(916) > 299) && (Zeny > 499)) { 
                        delitem 2247,1;  // Chapéu Romântico 
                        delitem 916,300;  // Plumas 
                        set Zeny,Zeny-500; 
                        mes "[Comerciante]"; 
                        mes "Ooh~! Você tem bom senso de moda. Eu sei que você teve um tempo difícil coletando esse material, mas este chapéu vale a pena. Pegue. Tudo que você precisa agora é de um casaco de pele e uma bengala!"; 
                        getitem 5018,1; // Chapéu com Pluma 
                        close; 
                } 
                else { 
                        mes "[Comerciante]"; 
                        mes "Você cara~"; 
                        mes "Verifique os requisitos do novo. Você não pareçe um idiota, mas. Então cara, caia na real."; 
                        close; 
                } 
        case 3: 
                if ((countitem(998) > 19) && (countitem(707) > 0) && (Zeny > 4999)){ //Iron Singing_Plant  
                        delitem 998,20;  // Ferro 
                        delitem 707,1;  // Planta da Melodia 
                        set Zeny,Zeny-5000; 
                        mes "[Comerciante]"; 
                        mes "Isso aqui? É um pouco assustador, embora ele tenha algum tipo de qualidade romântica. O que você acha? Você gosta disso? Tudo bem, pegue, é seu!"; 
                        getitem 2281,1; // Máscara do Fantasma 
                        close; 
                } 
                else { 
                        mes "[Comerciante]"; 
                        mes "Palhaço. Verifique os requisitos de novo. ^3355FFMáscara do Fantasma^000000 não é fácil chegar nisso. Então cara, caia na real."; 
                        close; 
                } 
        case 4: 
                if ((countitem(1019) > 119) && (Zeny > 9999)){  
                        delitem 1019,120;  // Tronco
                        set Zeny,Zeny-10000; 
                        mes "[Comerciante]"; 
                        mes "Se você tiver a oportunidade de visitar a Vila Uplander, Payon, visite e conheça o artesão do Chapéu Chinês."; 
                        next; 
                        mes "[Comerciante]"; 
                        mes "O Chapéu Chinês nunca foi vendido para comerciantes além de mim, já que apenas eu posso reconhecer a sua qualidade. Devido à sua raridade, o Chapéu Chinês se tornou um produto muito especial e excepcional. Tudo bem! Pegue, é seu!"; 
                        getitem 2280,1; // Chapéu Chinês 
                        close; 
                } 
                else { 
                        mes "[Comerciante]"; 
                        mes "MORON~ Verifique meus requisitos de novo. Então cara, Você não pareçe um idiota, caia na real~"; 
                        close; 
                } 
        } 
}
