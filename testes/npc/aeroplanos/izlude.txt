izlude,204,75,3	script	Equipe do Aeroporto#izlude	90,{

	mes "[Equipe do Aeroporto]";
	mes "Bem-vindo! Gostaria";
	mes "de embarcar no Dirigível";
	mes "e embarcar para os voos que vão até";
	mes "Juno e Rachel?";
	next;
	if (select("Embarcar no Dirigível para Juno/Rachel.:Cancelar.") == 1) {
		mes "[Equipe do Aeroporto]";
		mes "A taxa de embarque do Dirigível";
		mes "é 1,200 zeny, mas se você";
		mes "possuir um Bilhete para Aeroplano,";
		mes "a taxa não será cobrada. Você";
		mes "deseja embarcar?";
		next;
		if (select("Sim:Não") == 1) {
			if (countitem(7311) > 0)
				delitem 7311,1; // Bilhete para Aeroplano
      else if (Zeny >= 1200)
    		set Zeny, Zeny - 1200;
				warp "airplane_01",244,58;
				end;
			}
			mes "[Equipe do Aeroporto]";
			mes "Me desculpe, mas você não";
			mes "possui um Bilhete para";
			mes "Aeroplano e você não possui";
			mes "a quantidade de zenys suficientes";
			mes "para embarcar. Lembre, a";
			mes "taxa de embarque é 1,200 zeny.";
			close;
		}
	mes "[Equipe do Aeroporto]";
	mes "Obrigado e";
	mes "por favor volte novamente.";
	mes "Tenha um bom dia~";
	close;
}
